    <script>
        let currentPage = 1;

        function nextPage() {
            if (currentPage < 3) {
                document.getElementById('page' + currentPage).classList.remove('active');
                currentPage++;
                document.getElementById('page' + currentPage).classList.add('active');
            }
            if (currentPage === 3) {
                displaySummary();
            }
        }

        function displaySummary() {
            const summaryDiv = document.getElementById('summary');

            // Get qualifications
            const qualifications = document.querySelectorAll('input[name="qualification"]:checked');
            let qualificationList = [];
            qualifications.forEach(q => qualificationList.push(q.value));

            // Get personal details
            const name = document.getElementById('name').value;
            const age = document.getElementById('age').value;
            const weight = document.getElementById('weight').value;
            const height = document.getElementById('height').value;
            const medical = document.getElementById('medical').value;
            const contact = document.getElementById('contact').value;
            const address = document.getElementById('address').value;

            summaryDiv.innerHTML = `
                <p><strong>Name:</strong> ${name}</p>
                <p><strong>Age:</strong> ${age}</p>
                <p><strong>Weight:</strong> ${weight} kg</p>
                <p><strong>Height:</strong> ${height} cm</p>
                <p><strong>Medical Conditions:</strong> ${medical || "None"}</p>
                <p><strong>Contact:</strong> ${contact}</p>
                <p><strong>Address:</strong> ${address}</p>
                <p><strong>Selected Programs:</strong> ${qualificationList.join(', ')}</p>
            `;
        }

        function submitForm() {
            alert('Form submitted successfully!');
        }
    </script>
